# INSTRUCTIONS:
#
# condition				A diplomatic action can have any number of condition blocks, each with its own
#						tooltip, potential and allow section
#
# 	tooltip					Sets a custom text string similar to the hardcoded limits
# 							If no tooltip is scripted, the tooltip for the actual trigger will be shown
#							Note that the custom tooltip is only shown if the allow trigger is NOT met
#
# 	potential				Determines if the trigger is applicable or not
# 	allow					Determines if the action is valid or not
#
# effect				A diplomatic action can only have one effect block

# ROOT					actor
# FROM					target

declarewar = {
	condition = {
		tooltip = h_tooltip_cn_declarewar
		potential = {
			is_colonial_nation = yes
		}
		allow = {
			FROM = { same_continent = ROOT }
		}
	}
}

form_coalition = {
	condition = {
		tooltip = MINAGGRESSIVEEXPANSION
		potential = {	
		}
		allow = {
			has_opinion_modifier = {
				modifier = aggressive_expansion
				who = FROM
			}		
			NOT = {
				has_opinion_modifier = {
					modifier = aggressive_expansion
					who = FROM
					value = -50
				}
			}
		}
	}
}

request_to_join_trade_league = {
	condition = {
		tooltip = NOT_RELEVANT_TRADE_REGION_FROM
		potential = {
		}
		allow = {
			home_trade_node = {
				trade_share = {
					country = from
					share = 1
				}
			}
		}
	}
}

allianceaction = {
	condition = {
		tooltip = h_tooltip_cn_ally
		potential = {
			is_colonial_nation = no
		}
		allow = {
			FROM = { is_colonial_nation = no }
		}
	}
	condition = {
		tooltip = h_tooltip_many_allies
		potential = {
		}
		allow = {
			ROOT = {
				NOT = {
					num_of_allies = 5
				}
			}
		}
	}
	condition = {
		tooltip = h_tooltip_from_many_allies
		potential = {
		}
		allow = {
			FROM = {
				NOT = {
					num_of_allies = 5
				}
			}
		}
	}
	condition = {
		tooltip = sm_tooltip_mamottoally
		potential = {
			OR = {
				tag = MAM
				tag = EGY
			}	
			NOT = {
				has_reform = parliamentary_reform
				has_reform = royal_decree_reform
				has_reform = aristocratic_court_reform
				has_reform = general_estates_reform
				has_reform = states_general_reform
				has_reform = legislative_sejm
				has_reform = integrated_sejmiks
				has_reform = feuillant_reform
				has_reform = two_chamber_reform
				has_reform = imperial_decree_reform
			}
		}	
		allow = {
			FROM = {
				NOT = { 
					OR = {
						tag = TUR					
					}
				}
			}
		}
	}
	
	condition = {
		tooltip = sm_tooltip_ottomamally
		potential = {
			AND = {
				tag = TUR
				has_reform = ottoman_government
			}
		}	
		allow = {
			FROM = {
				NOT = { 
					OR = {
						tag = MAM
						tag = EGY						
					}
				}
			}
		}
	}
}
