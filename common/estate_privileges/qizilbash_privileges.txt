estate_qizilbash_levies = {
        icon = privilege_demand_more_troops
		land_share = 5
        max_absolutism = -5        
        loyalty = -0.05               
        influence  = 0.05            
 	is_valid = { 
		any_owned_province = {
			OR = {
				culture = azerbaijani
				#culture = mazandarani
				#culture = kurdish
				culture = luri
				culture = turkish
				}
			}
		}	
	can_select = {
		NOT = {
			num_of_estate_privileges = {
			estate = estate_qizilbash
			value = 4
			}
		}
	}
	on_granted = {
		custom_tooltip = estate_qizilbash_levies_effect_tooltip
		hidden_effect = {
			every_owned_province = {
				limit = {
					OR = {
					culture = azerbaijani
					#culture = mazandarani
					#culture = kurdish
					culture = luri
					culture = turkish
					}
				}
				add_province_modifier = {
					name = estate_qizilbash_levies_mod
					duration = -1
				}
			}
		}
	}
	on_revoked = {
		custom_tooltip = estate_qizilbash_levies_revoke_effect_tooltip
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_province_modifier = estate_qizilbash_levies_mod
				}
				remove_province_modifier = estate_qizilbash_levies_mod
			}
		}
	}
	on_invalid = {
		custom_tooltip = estate_qizilbash_levies_revoke_effect_tooltip
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_province_modifier = estate_qizilbash_levies_mod
				}
				remove_province_modifier = estate_qizilbash_levies_mod
			}
		}
	}	
	ai_will_do = {					
        factor = 1000                               
        }
}
